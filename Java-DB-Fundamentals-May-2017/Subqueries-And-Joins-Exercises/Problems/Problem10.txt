SELECT 
    e1.employee_id,
    CONCAT_WS(' ', e1.first_name, e1.last_name) AS 'employee_name',
    CONCAT_WS(' ', e2.first_name, e2.last_name) AS 'manager_name',
    d.name AS 'department_name'
FROM
    employees AS e1
        INNER JOIN
    employees AS e2 ON e1.manager_id = e2.employee_id
        INNER JOIN
    departments AS d ON e1.department_id = d.department_id
ORDER BY e1.employee_id ASC 
LIMIT 5
